%Create a new minipage environment where paragraphs have indents
% minipage is convenient for placing an image next to a text, without needing to deal with wrapfig
% but inside minipage the paragraphs start with no indent, therefore this new environment
% taken from:
% https://tex.stackexchange.com/a/619781/173645
\newlength{\currentparindent} % define new variable
% \newenvironment{name}[numargs][optarg_default]{begin_code}{end_code}
\newenvironment{minipageparindent}[2][c]{
    \setlength{\currentparindent}{\parindent} % save the value
    \noindent\begin{minipage}[#1]{#2} % open the minipage
    \setlength{\parindent}{\currentparindent} % restore the value
    }{\end{minipage}} % close the minipage
